Tests p:string-replace

Test ab-string-replace-001.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:input port="source"/>
         <p:output port="result"/>
            
         <p:string-replace match="doc/@version">
            <p:with-option name="replace" select="concat('"', number(/doc/@version)+1,'"')"/>
         </p:string-replace>   
      </p:declare-step>

Inputs

port = source


      <doc version="1">
         <para>some text</para>
      </doc>

Schematron validation


      <s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron">
         <s:pattern>
            <s:rule context="/">
               <s:assert test="doc">Root element is not 'doc'.</s:assert>
               <s:assert test="doc/@version='2'">Root element does not have @version = 2.</s:assert>
               <s:assert test="doc/para">Root element does not have child "para".</s:assert>
               <s:assert test="doc/para/text()='some text'">Element 'para' does not have a text child 'some text'.</s:assert>
            </s:rule>
         </s:pattern>
      </s:schema>
