Option declaration 037 (AB)

Checks that a short cut for an array is interpreted as XPath expression.

Test ab-option-037.xml is expected to pass.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" xmlns:test="http://xproc.org/ns/testsuite/3.0/demo" xmlns:xs="http://www.w3.org/2001/XMLSchema" version="3.0">
   <p:output port="result">
   <p:declare-step type="test:test">
      <p:output port="result">
      <p:option name="option" as="array(*)" required="true">
      <p:identity>
         <p:with-input>
            <result>{$option?1}</result>
         </p:with-input>
      </p:identity>
   </p:declare-step>
   <test:test option="['value', 'something else']">
</p:declare-step>

Schematron validation


<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="result">The pipeline root is not result.</s:assert>
         <s:assert test="result/text()='value'">The text child of root element is not 'value'.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
