AB-xslt-109

Tests nodes are correctly returned in arrays/maps. (Thanks to Martin for the test's blueprint).

Test ab-xslt-109.xml is expected to pass.

It requires the following features: xslt-3.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         
         <p:xslt version="3.0" template-name="start" name="xslt">
            <p:with-input port="source"><p:empty/></p:with-input>
            <p:with-input port="stylesheet" expand-text="false">
               <xsl:stylesheet xmlns:xsl="http://www.w3.org/1999/XSL/Transform" version="3.0">
                  <xsl:output build-tree="no"/>
                  <xsl:template name="start" as="array(comment())">
                     <xsl:variable name="document">
                        <doc xmlns="">
                           <one><xsl:comment select="'comment'"/></one>
                           <two><xsl:comment select="'comment'"/></two>
                        </doc>
                     </xsl:variable>
                     <xsl:sequence select="array{ $document/doc/*/comment() }"/>
                  </xsl:template>
               </xsl:stylesheet>
            </p:with-input>
         </p:xslt>
         <p:identity>
            <p:with-input>
               <result xmlns="">
                  <array-check>{. instance of array(*)}</array-check>
                  <array-size-check>{array:size(.)}</array-size-check>
                  <array-of-nodes-check>{. instance of array(node())}</array-of-nodes-check>
                  <array-of-comment-check>{. instance of array(comment())}</array-of-comment-check>
                  <array-item-is-comment-check>{every $item in .?* satisfies $item instance of comment()}</array-item-is-comment-check> 
               </result>
            </p:with-input>
         </p:identity>
      </p:declare-step>
