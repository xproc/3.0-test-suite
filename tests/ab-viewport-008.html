viewport-008 (AB)

Test p:viewport with p:output that is bound to a sequence of documents, mixing the inline and pipe bindings.

Test ab-viewport-008.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         <p:viewport match="para" name="viewport">
            <p:with-input>
               <doc xmlns="">
                  <para>Some paragraph.</para>
                  <para>Some paragraph.</para>
                  <para>Some paragraph.</para>
                  <para>
                     <para>Nested paragraph.</para>
                  </para>
               </doc>
            </p:with-input>    
            <p:output port="tmp" sequence="true">
               <p:inline><foo xmlns=""/></p:inline>
               <p:pipe step="identity1" port="result"/>
            </p:output>
            <p:identity name="identity1">
               <p:with-input port="source">
                  <p:inline><bar xmlns=""/></p:inline>
               </p:with-input>
            </p:identity>
            <p:identity name="identity2">
               <p:with-input port="source">
                  <p:inline><baz xmlns=""/></p:inline>
               </p:with-input>
            </p:identity>
            <p:sink/>
         </p:viewport>
      </p:declare-step>
