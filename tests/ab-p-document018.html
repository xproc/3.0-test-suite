p:document 018

Tests p:document with a simple text document in ISO-8859-1 (charset specified in content-type).

Test ab-p-document018.xml is expected to pass.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result"/>
   <p:identity>
      <p:with-input>
         <p:document href="../documents/text-file-iso-8859-1.txt" content-type="text/plain; charset=iso-8859-1"/>
      </p:with-input>
   </p:identity>
   <p:wrap-sequence wrapper="doc"/>
</p:declare-step>

Schematron validation


<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="doc">The document root is not doc.</s:assert>
         <s:assert test="doc/text() = 'This is a simple text file in ISO-8859-1: ä ö ü'">Text is not correct.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
