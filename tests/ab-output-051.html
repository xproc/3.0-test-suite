Tests that documents appear in the right order on p:output.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
        <p:output port="result"/>
         
        <p:try>
           <p:identity>
              <p:with-input><primary xmlns=""/></p:with-input>
           </p:identity>
           <p:catch code="error">
              <p:identity>
                 <p:with-input><error xmlns=""/></p:with-input>
              </p:identity>
           </p:catch>
           <p:finally>
              <p:output port="additional" sequence="true" primary="false">
                 <p:pipe step="producer"/>
                 <p:inline><doc1 xmlns=""/></p:inline>
              </p:output>
              <p:identity name="producer">
                 <p:with-input><doc2 xmlns=""/></p:with-input>
              </p:identity>
           </p:finally>  
        </p:try>
        <p:wrap-sequence wrapper="result">
           <p:with-input pipe="additional"/>
        </p:wrap-sequence>
      </p:declare-step>

Schematron validation

            <s:rule context="/">
               <s:assert test="result">The document root is not 'result'.</s:assert>
               <s:assert test="result/*[1]/name()='doc2'">The first child element of result is not named doc2.</s:assert>
               <s:assert test="result/*[2]/name()='doc1'">The second child element of result is not named doc1.</s:assert>
            </s:rule>
         </s:pattern>
      </s:schema>
