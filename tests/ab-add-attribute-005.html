Add-attribute-005 (AB)

Tests adding an attribute with name in p:with-option to doc.

Test ab-add-attribute-005.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         <p:add-attribute match="/doc" attribute-value="5">
            <p:with-input>
               <doc xmlns=""/>
            </p:with-input>
            <p:with-option name="attribute-name" select="'att'"/>
         </p:add-attribute>
      </p:declare-step>

Schematron validation


      <s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
         <s:pattern>
            <s:rule context="/">
               <s:assert test="doc">The document root is not doc.</s:assert>
               <s:assert test="doc/@att">There is not attribute "att" at doc.</s:assert>
               <s:assert test="doc/@att=5">Value of attribute "att" is not 5.</s:assert>
            </s:rule>
         </s:pattern>
      </s:schema>
