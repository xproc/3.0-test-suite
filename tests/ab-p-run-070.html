The pipeline

   <p:output port="result"/>
   <p:option name="opt" select="'pipeline'"/>
   <p:run>
      <p:with-input>
         <p:inline expand-text="false">
            <p:declare-step version="3.0">
               <p:output port="result"/>
               <p:option name="opt" as="xs:string" required="true"/>

                  <p:with-input>
                     <result>{$opt}</result>
                  </p:with-input>
               </p:identity>
            </p:declare-step>
         </p:inline>
      </p:with-input>
      <p:run-option name="opt" select="'run-option'"/>
      <p:output port="result"/>
   </p:run>
</p:declare-step>
