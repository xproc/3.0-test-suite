Step available 001 (NW)

Test a user-defined step with a body that is conditionally present.

Test nw-step-available-001.xml is expected to pass.

The pipeline


<p:declare-step xmlns:ex="http://example.com/ns/xproc-extensions" xmlns:p="http://www.w3.org/ns/xproc" name="main" version="3.0">
   <p:output port="result"/>
   <p:declare-step type="ex:step1">
      <p:input port="source"/>
      <p:output port="result"/>
      <p:identity use-when="p:step-available('ex:step2')"/>
   </p:declare-step>
   <p:declare-step type="ex:step2">
      <p:input port="source"/>
      <p:output port="result"/>
   </p:declare-step>
   <p:identity>
      <p:with-input>
         <available>{p:step-available('ex:step1')}</available>
      </p:with-input>
   </p:identity>
</p:declare-step>

<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">

      <s:rule context="/">
         <s:assert test="available = 'false'">The step should not be available.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
