Static variables/option 047

Tests that a static variable is not visible in a sibling step.

Test ab-static-option-var-047.xml is expected to fail with error code err:XD0050.

The pipeline


        <declare-step xmlns:p="http://www.w3.org/ns/xproc" name="main" version="3.0">
            <output port="result"/>
            
            <declare-step type="dummy:test1">
                <output port="result"/>
                <variable static="true" name="static-variable" select="15"/>
                
                <identity name="test1-one">
                    <with-input><doc xmlns="">{$static-variable}</doc></p:with-input>
                </p:identity>
            </p:declare-step>
            
            <declare-step type="dummy:test2">
                <output port="result"/>
                
                <identity name="test2-one">
                    <with-input><doc xmlns="">{$static-variable}</doc></p:with-input>
                </p:identity>
            </p:declare-step>
            
            <dummy:test1 xmlns:dummy="http://dummy" name="one"/> 
            <dummy:test2 xmlns:dummy="http://dummy" name="two"/>
            
            <wrap-sequence wrapper="result">
                <with-input pipe="@one @two"/>
            </p:wrap-sequence>
            
        </p:declare-step>
