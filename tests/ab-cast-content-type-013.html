cast-content-type 013 (AB)

Tests casting an HTML document to an xml content-type.

Test ab-cast-content-type-013.xml is expected to pass.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" name="pipeline" version="3.0">
   <p:output port="result">
   <p:cast-content-type content-type="application/xml" name="caster">
      <p:with-input>
         <p:inline content-type="text/html" document-properties="map{'base-uri' : 'http://xproc-test.org', 'serialization' : map{'indent' : true()}, 'additional' : 'property'}">
            <html/>
         </p:inline>
      </p:with-input>
   </p:cast-content-type>
   <p:cast-content-type content-type="application/xml" name="properties">
      <p:with-input select="p:document-properties(.)">
   </p:cast-content-type>
   <p:wrap-sequence wrapper="result">
      <p:with-input pipe="@caster @properties">
   </p:wrap-sequence>
</p:declare-step>

Schematron validation


<s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:ns prefix="fn" uri="http://www.w3.org/2005/xpath-functions">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="result">The root element is not result.</s:assert>
         <s:assert test="result/html">Root element does not have child 'html'.</s:assert>
         <s:assert test="result/fn:map">Root element does not have child 'fn:map'.</s:assert>
         <s:assert test="result/fn:map/fn:string[@key='base-uri']='http://xproc-test.org'">The base-uri property is not correct.</s:assert>
         <s:assert test="result/fn:map/fn:string[@key='content-type']='application/xml'">The content-type property is not correct.</s:assert>
         <s:assert test="result/fn:map/fn:string[@key='additional']='property'">The additional property is not correct.</s:assert>
         <s:assert test="not(result/fn:map/fn:map[@key='serialization'])">There should be no serialization property.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
