The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result">
   <p:variable name="episode" select="p:system-property('p:episode')">
   <p:file-create-tempfile delete-on-exit="true">
   <p:variable name="path" select="*:result/text()">
   <p:try name="identity">
      <p:add-attribute match="/*" attribute-name="att" depends="storer">
         <p:with-option name="attribute-value" select="doc/text()">
            <p:document href="{$path}" content-type="application/xml">
         </p:with-option>
         <p:with-input>
            <doc/>
         </p:with-input>
      </p:add-attribute>
      <p:catch>
         <p:identity>
            <p:with-input>
               <failure/>
            </p:with-input>
         </p:identity>
      </p:catch>
   </p:try>
   <p:sink/>
   <p:store href="{$path}" name="storer">
      <p:with-input>
         <doc>{$episode}</doc>
      </p:with-input>
   </p:store>
   <p:choose>
      <p:with-input pipe="@identity">
      <p:when test="doc/@att = $episode">
         <p:identity>
            <p:with-input>
               <success/>
            </p:with-input>
         </p:identity>
      </p:when>
      <p:otherwise>
         <p:identity>
            <p:with-input>
               <failure/>
            </p:with-input>
         </p:identity>
      </p:otherwise>
   </p:choose>
</p:declare-step>
