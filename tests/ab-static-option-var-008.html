Static variables/option 008

Tests that a static option can be used in avt of dynamic inline.

Test ab-static-option-var-008.xml is expected to pass.

The pipeline


        <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" name="main" version="3.0">
            <p:output port="result"/>
            <p:option static="true" name="static-option" select="18"/>

            <p:cast-content-type content-type="application/xml">
                <p:with-input select="p:document-properties(.)">
                    <p:inline document-properties="map{'a': $static-option}">
                        <doc xmlns="http://www.w3.org/1999/xhtml" xmlns="http://www.w3.org/1999/xhtml"/>
                    </p:inline>
                </p:with-input>
            </p:cast-content-type>
        
        </p:declare-step>

Schematron validation


        <s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron">
            <s:ns prefix="j" uri="http://www.w3.org/2005/xpath-functions"/>
            
            <s:pattern>
                <s:rule context="/">
                    <s:assert test="j:map">The pipeline root is not “map”.</s:assert>
                    <s:assert test="j:map/j:number[@key='a']/text()='18'">Entry for 'a' is not '18'.</s:assert>
                </s:rule>
            </s:pattern>
        </s:schema>
