AB-xslt-103

Tests @populate-default-collection is correctly handled.

Test ab-xslt-103.xml is expected to pass.

It requires the following features: xslt-2.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result">
   <p:xslt version="2.0" populate-default-collection="false" name="xslt">
      <p:with-input port="source">
         <a/>
         <b/>
         <c/>
         <d/>
      </p:with-input>
      <p:with-input port="stylesheet">
         <xsl:stylesheet xmlns:xsl="http://www.w3.org/1999/XSL/Transform" version="2.0">
            <xsl:template match="a">
               <result>
                  <xsl:try>
                     <xsl:value-of select="count(collection())">
                     <xsl:catch>
                        <not-defined/>
                     </xsl:catch>
                  </xsl:try>
               </result>
            </xsl:template>
         </xsl:stylesheet>
      </p:with-input>
   </p:xslt>
</p:declare-step>

Schematron validation


<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="/result">The document root is not 'result'.</s:assert>
         <s:assert test="/result/not-defined">Root element does not have a child 'not-defined'.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
