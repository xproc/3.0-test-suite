AB-xslt-135

Testing for node identity in p:xslt (2.0).

Test ab-xslt-135.xml is expected to pass.

It requires the following features: xslt-2.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result"/>
   <p:identity name="producer">
      <p:with-input>
         <doc/>
      </p:with-input>
   </p:identity>
   <p:xslt version="2.0">
      <p:with-input pipe="@producer @producer"/>
      <p:with-input port="stylesheet">
         <xsl:stylesheet xmlns:xsl="http://www.w3.org/1999/XSL/Transform" version="2.0">
            <xsl:template match="/">
               <result>
                  <xsl:value-of select="collection()[1] is collection()[2]"/>
               </result>
            </xsl:template>
         </xsl:stylesheet>
      </p:with-input>
   </p:xslt>
</p:declare-step>

Schematron validation


<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="/result">The document root is not 'result'.</s:assert>
         <s:assert test="/result/text()='true'">Test for required node identity fails.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
