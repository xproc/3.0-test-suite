Collection 002

Tests that err:XD0001 is raised if a sequence of documents is the context for an XPath expression on a p:variable.

Test nw-collection-002.xml is expected to fail with error code err:XD0001.

      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
        <p:output port="result"/>

        <p:identity name="id1">
          <p:with-input port="source">
            <p:inline>
              <doc xmlns=""/>
            </p:inline>
          </p:with-input>
        </p:identity>

        <p:identity name="id2">
          <p:with-input port="source">
            <p:inline>
              <doc xmlns=""/>
            </p:inline>
          </p:with-input>
        </p:identity>

        <p:identity name="combine">
          <p:with-input port="source" pipe="@id1 @id2"/>

        <p:variable name="a" select="count(.)"/>

        <p:identity>
          <p:with-input>
            <doc xmlns="">{$a}</doc>
          </p:with-input>
        </p:identity>
      </p:declare-step>
