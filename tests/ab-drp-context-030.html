AB-context-p:finally-002

Tests port "error" is context node for TVT in finally.

Test ab-drp-context-030.xml is expected to pass.

The pipeline


<p:declare-step xmlns:c="http://www.w3.org/ns/xproc-step" xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result">
   <p:try name="try">
      <p:error code="error">
         <p:with-input>
            <ERROR>failure</ERROR>
         </p:with-input>
      </p:error>
      <p:catch>
         <p:identity>
            <p:with-input>
               <irrelevant/>
            </p:with-input>
         </p:identity>
      </p:catch>
      <p:finally>
         <p:output port="finally" primary="false" pipe="@producer">
         <p:identity name="producer">
            <p:with-input>
               <p:inline>
                  <result>{/c:errors/c:error/ERROR/text()}</result>
               </p:inline>
            </p:with-input>
         </p:identity>
      </p:finally>
   </p:try>
   <p:identity>
      <p:with-input pipe="finally@try">
   </p:identity>
</p:declare-step>
