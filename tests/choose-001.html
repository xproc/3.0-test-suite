The pipeline

<declare-step xmlns:p="http://www.w3.org/ns/xproc" name="main" version="3.0">
  <option name="match"/>
  <output port="result"/>

  <identity name="id">
    <with-input port="source">
      <inline document-properties="map { 'a': '1', 'b': '2' }">
        <doc/>
      </p:inline>
    </p:with-input>
  </p:identity>


  <choose name="choose">
    <when test="p:document-property(., 'a') = $match">
      <identity>
        <with-input port="source">
          <inline><doc>one</doc></p:inline>
        </p:with-input>
      </p:identity>
    </p:when>
    <when test="p:document-property(., 'b') = $match">
      <identity>
        <with-input port="source">
          <inline><doc>two</doc></p:inline>
        </p:with-input>
      </p:identity>
    </p:when>
    <otherwise>
      <identity>
        <with-input port="source">
          <inline><doc>none</doc></p:inline>
        </p:with-input>
      </p:identity>
    </p:otherwise>
  </p:choose>

</p:declare-step>
