Static variables/option 025

Tests that a static option is not visible in a parent step.

Test ab-static-option-var-025.xml is expected to fail with error code err:XS0107.

The pipeline


<p:declare-step xmlns:dummy="http://dummy" xmlns:err="http://www.w3.org/ns/xproc-error" xmlns:p="http://www.w3.org/ns/xproc" name="main" version="3.0">
   <p:output port="result">
   <p:declare-step type="dummy:test1">
      <p:output port="result">
      <p:option static="true" name="static-option" select="15">
      <p:identity name="test1-one">
         <p:with-input>
            <doc>{$static-option}</doc>
         </p:with-input>
      </p:identity>
   </p:declare-step>
   <dummy:test1 name="one">
   <p:identity name="two">
      <p:with-input>
         <doc>{$static-option}</doc>
      </p:with-input>
   </p:identity>
   <p:wrap-sequence wrapper="result">
      <p:with-input pipe="@one @two">
   </p:wrap-sequence>
</p:declare-step>
