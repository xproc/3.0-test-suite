Schematron validation


<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:pattern>
      <s:rule context="/">
         <s:assert test="wrapper">The document root is not wrapper.</s:assert>
         <s:assert test="wrapper/doc[1]/title='Some title'">doc[1]/title is not 'Some title'.</s:assert>
         <s:assert test="wrapper/doc[1]/para='Some paragraph.'">doc[1]/para is not 'Some paragraph.'</s:assert>
         <s:assert test="wrapper/doc[2]/title='Some other title'">doc[2]/title is not 'Some other title'.</s:assert>
         <s:assert test="wrapper/doc[2]/para='Some other paragraph.'">doc[2]/para is not 'Some other paragraph.'</s:assert>
         <s:assert test="count(wrapper/doc)=2">Root 'wrapper' does not have two 'doc'-children.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
