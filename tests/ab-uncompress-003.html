AB-uncompress-003

Tests document-properties uncompress result.

Test ab-uncompress-003.xml is expected to pass.

The pipeline


<p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
   <p:output port="result"/>
   <p:uncompress content-type="application/xml" format="gzip">
      <p:with-input>
         <p:document href="../documents/simpledoc.gz" document-properties="map{'additional' : 'bogus'}"/>
      </p:with-input>
   </p:uncompress>
   <p:cast-content-type content-type="application/xml">
      <p:with-input select="p:document-properties(.)"/>
   </p:cast-content-type>
</p:declare-step>

<s:schema xmlns="http://www.w3.org/1999/xhtml" xmlns:s="http://purl.oclc.org/dsdl/schematron" queryBinding="xslt2">
   <s:ns prefix="j" uri="http://www.w3.org/2005/xpath-functions"/>
   <s:pattern>
      <s:rule context="/">
         <s:assert test="j:map">The document root is not 'j:map'.</s:assert>
         <s:assert test="j:map/j:string[@key='content-type']/text()='application/xml'">Property 'content-type' is not 'application/xml'.</s:assert>
         <s:assert test="ends-with(j:map/j:string[@key='base-uri']/text(),'simpledoc.gz')">Property 'base-uri' does not end with 'simpledoc.gz'.</s:assert>
         <s:assert test="j:map/j:string[@key='additional']/text()='bogus'">Property 'additional' is not 'bogus'.</s:assert>
      </s:rule>
   </s:pattern>
</s:schema>
