viewport-005 (AB)

Test p:viewport with p:iteration-position() and p:iteration-size().

Test ab-viewport-005.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         <p:viewport match="para" name="viewport">
            <p:with-input>
               <doc xmlns="">
                  <para>Some paragraph.</para>
                  <para>Some paragraph.</para>
                  <para>Some paragraph.</para>
                  <para>
                     <para>Nested paragraph.</para>
                  </para>
               </doc>
            </p:with-input>    
            <p:choose name="choose">
               <p:when test="p:iteration-position() = 1">
                  <p:identity>
                     <p:with-input>
                        <p:inline><first xmlns=""/></p:inline>
                        <p:pipe step="viewport" port="current"/>
                     </p:with-input>
                  </p:identity>
               </p:when>
               <p:when test="p:iteration-position() = p:iteration-size()">
                  <p:identity>
                     <p:with-input>
                        <p:pipe step="viewport" port="current"/>
                        <p:inline><last xmlns=""/></p:inline>
                     </p:with-input>
                  </p:identity>
               </p:when>
               <p:otherwise>
                  <p:identity/>
               </p:otherwise>
            </p:choose>
            
         </p:viewport>
      </p:declare-step>

             </s:rule>
         </s:pattern>
