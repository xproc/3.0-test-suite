Tests a p:variable is explicitly connectable to the containers primary input port (using just @step) .

Test ab-variable-040.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" name="pipeline" version="3.0">
            <p:output port="result"/>
            <p:input port="source"/>
            
            <p:variable name="var" select="/doc/text()">
                <p:pipe step="pipeline"/>
            </p:variable>
            <p:identity>
                <p:with-input port="source">
                    <doc xmlns="">{$var}</doc>
                </p:with-input>
            </p:identity>

        </p:declare-step>
