AB-namespace-delete-003

Removing default namespace does not harm elements/attribute in other namespace.

Test ab-namespace-delete-003.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         
         <p:namespace-delete prefixes="pre">
            <p:with-input>
               <doc xmlns="http://dummy" xmlns="http://dummy">
                  <section xmlns:other="http://other" xmlns="http://dummy" other:id="one">
                     <other:par xmlns:other="http://other" xmlns="http://dummy">This is a para.</other:par>
                     <other:par xmlns:other="http://other" xmlns="http://dummy">This is another para.</other:par>
                  </section>
                  <section xmlns:other="http://other" xmlns="http://dummy" other:id="two">
                     <other:par xmlns:other="http://other" xmlns="http://dummy">First para.</other:par>
                     <other:par xmlns:other="http://other" xmlns="http://dummy">Second para.</other:par>
                  </section>
               </doc>
            </p:with-input>
         </p:namespace-delete>
         
      </p:declare-step>

Schematron validation


      <s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron">
         <s:ns prefix="other" uri="http://other"/>
         <s:pattern>
            <s:rule context="/">
               <s:assert test="/doc">The document root is not 'doc'.</s:assert>
               <s:assert test="count(/doc/section)=2">Element doc does not have two children 'section'.</s:assert>
            </s:rule>
            <s:rule context="/doc/section">
               <s:assert test="./@other:id">Section does not have an attribute 'id'.</s:assert>
               <s:assert test="count(other:par)=2">Section does not have two paragraphs.</s:assert>
            </s:rule>
         </s:pattern>
      </s:schema>
