viewport-019 (AB)

Test p:viewport with matched document node replace by text document to be text document.

Test ab-viewport-019.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
         <p:output port="result"/>
         <p:viewport match="/">
            <p:with-input>
               <p:inline document-properties="map{'serialization' : map{'indent': true()}, 'prop':'bonus'}">
                  <doc xmlns="">
                     <para>Some para.</para>
                  </doc>
               </p:inline>
            </p:with-input>    
            <p:identity>
               <p:with-input port="source">
                  <p:inline content-type="text/plain">Some new text.</p:inline>
               </p:with-input>
            </p:identity>
         </p:viewport>
         <p:identity>
            <p:with-input select="p:document-properties-document(.)"/>
         </p:identity>
      </p:declare-step>

Schematron validation


      <s:schema xmlns:s="http://purl.oclc.org/dsdl/schematron">
         <s:ns prefix="c" uri="http://www.w3.org/ns/xproc-step"/>
         <s:pattern>
            <s:rule context="/">
               <s:assert test="c:document-properties">The document root is not 'c:document-properties'.</s:assert>
               <s:assert test="c:document-properties/content-type='text/plain'">Document's content-type is not text/plain.</s:assert>
               <s:assert test="not(c:document-properties/serialization)">Serialization properties are not removed.</s:assert>
               <s:assert test="c:document-properties/prop">Extra property 'prop' is not preserved.</s:assert>
            </s:rule>
          </s:pattern>
      </s:schema>
