Test ab-p-archive-030.xml is expected to pass.

The pipeline


      <p:declare-step xmlns:p="http://www.w3.org/ns/xproc" version="3.0">
            <p:output port="result"/>        

         <p:archive parameters="map{'method': 'none'}">
               <p:with-input port="source"><p:empty/></p:with-input>
               <p:with-input port="manifest">
                  <c:archive xmlns:c="http://www.w3.org/ns/xproc-step">
                     <c:entry name="one.xml" href="../documents/ab-doc.xml" method="deflated"/>
                     <c:entry name="two.xml" href="../documents/ab-doc2.xml"/>
                  </c:archive>
               </p:with-input>
               <p:with-input port="archive"><p:empty/></p:with-input>
            </p:archive>
 
            <p:archive-manifest relative-to="http://xproc.org/ns/testsuite"/>
      </p:declare-step>
