   <p:output port="result"/>

      <p:with-input port="source">
         <p:inline document-properties="map { 'a': '1', 'b': '2' }">

               </p:inline>
            </p:with-input>
         </p:identity>
      </p:when>
      <p:when test="p:document-property(., 'b') = $match">
         <p:identity>
            <p:with-input port="source">

                  <doc>two</doc>

            </p:with-input>
         </p:identity>
      </p:when>
      <p:otherwise>
         <p:identity>
            <p:with-input port="source">
               <p:inline>
                  <doc>none</doc>
               </p:inline>

         </p:identity>
      </p:otherwise>
   </p:choose>
